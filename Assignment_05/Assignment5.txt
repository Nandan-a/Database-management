*MongoDB*

* Create Database
  use db_name;
* Create Collections
  db.createCollection("collection_name")
* Insert one document 
  db.db_name.insertOne({Key: Value});
* Insert multiple documents
  db.db_name.insertMany({Key:Value},{Key:Value});
* Read/Display data from collections
  db.db_name.find({Key: Value})
  db.db_name.findOne({Key: Value})
* Update values in collections
  db.db_name.updateOne({Key: Value}, { $set: {Key: Value} })
  db.db_name.updateMany({Key: Value}, { $set: {Key: Value} })
* Remove documnets
  db.db_name.deleteOne({ Key: Value })
  db.db_name.deleteMany({ Key: Value })
* Remove Collections
  db.db_name.drop()
